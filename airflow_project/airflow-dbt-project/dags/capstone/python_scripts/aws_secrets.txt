import boto3
import ast
import os

def get_secret(
    key: str,
    secret: str,
    region: str,
    secret_name: str
    ):


    session = boto3.session.Session(aws_access_key_id=key, aws_secret_access_key=secret, region_name=region)
    client = session.client(service_name='secretsmanager')

    secret_val = client.get_secret_value(SecretId=secret_name)

    return secret_val
