from airflow.decorators import dag
from airflow import DAG
from airflow.operators.python import PythonOperator
from datetime import datetime, timedelta
import sys
import os
from dags.capstone.python_scripts.load_treasury_into_snowflake import load_treasury_main 

default_args = {
    'owner': 'andres',
    'depends_on_past': False,
    'retries': 1,
    'retry_delay': timedelta(minutes=5),
    'email_on_failure': False,
    "start_date": datetime(2024, 11, 1),
    'email_on_retry': False,
    'execution_timeout': timedelta(hours=1),
    'wait_for_downstream':True
}


# Define the DAG.
@dag(
    default_args=default_args,
    description='DAG to load Treasury data into Snowflake using Snowpark',
    start_date=datetime(2024, 11, 1),
    # schedule_interval="@daily",
    schedule_interval="0 1 * * *",
    catchup=True,
    template_searchpath='dags/capstone/python_scripts',
    max_active_runs=3,
)   


    
def load_treasury_to_snowflake_dag():
    load_data_task = PythonOperator(
    task_id='load_data_task',
    python_callable=load_treasury_main,
    depends_on_past=True,
    
    # dag=dag,
    op_kwargs={
        "date_to_ingest": '{{ yesterday_ds }}'
    },
    )

# Set task dependencies if you have more tasks
    load_data_task

load_treasury_to_snowflake_dag()
